Extract and plot the top topics of the selected column.

>>> from techminer2 import *
>>> directory = "/workspaces/techminer2/data/"
>>> file_name = "/workspaces/techminer2/sphinx/images/top_topics.png"
>>> top_topics(
...     column="author_keywords",
...     directory=directory,
... ).savefig(file_name)

>>> top_topics(
...     column='author_keywords',
